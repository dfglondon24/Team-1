    <script>
        const urlParams = new URLSearchParams(window.location.search);
        const studentIndex = urlParams.get('index');
        const students = JSON.parse(localStorage.getItem('students')) || [];

        if (studentIndex !== null && students[studentIndex]) {
            const student = students[studentIndex];

            // Display student details
            document.getElementById('studentDetails').innerHTML = `
                <p><strong>Name:</strong> ${student.name}</p>
                <p><strong>Age:</strong> ${student.age}</p>
                <p><strong>Community:</strong> ${student.community}</p>
                <p><strong>District:</strong> ${student.district}</p>
                <p><strong>Region:</strong> ${student.region}</p>
                <p><strong>Education Level:</strong> ${student.education}</p>
                <p><strong>Currently in Education:</strong> ${student.inEducation ? 'Yes' : 'No'}</p>
                <p><strong>Additional Notes:</strong> Chest infection January 2023, out of school for 2 months.</p>
            `;
        } else {
            document.getElementById('studentDetails').innerHTML = '<p>No student information found.</p>';
        }
    </script>

    <h2>Numeracy and Literacy Scores Over Time</h2>
    <canvas id="scoreChart"></canvas> <!-- This is the canvas for the chart -->

    <script>
        // Hard-coded data for the chart
        const dates = ['2023-01-01', '2023-03-01', '2023-05-01', '2023-07-01', '2023-09-01'];
        const literacyScores = [65, 70, 75, 80, 85];
        const numeracyScores = [60, 65, 70, 72, 78];

        // Create chart for numeracy and literacy scores over time
        const ctx = document.getElementById('scoreChart').getContext('2d');
        new Chart(ctx, {
            type: 'line', // Simple line chart
            data: {
                labels: dates, // X-axis labels (dates)
                datasets: [
                    {
                        label: 'Literacy Score',
                        data: literacyScores, // Literacy data
                        borderColor: 'rgba(75, 192, 192, 1)', // Line color
                        fill: false // No background fill
                    },
                    {
                        label: 'Numeracy Score',
                        data: numeracyScores, // Numeracy data
                        borderColor: 'rgba(153, 102, 255, 1)', // Line color
                        fill: false // No background fill
                    }
                ]
            },
            options: {
                scales: {
                    y: {
                        beginAtZero: true // Make sure the chart starts at 0
                    }
                },
                responsive: true
            }
        });
    </script>
